  --spacing-none: 0;

  --spacing-small: .5rem;

  --spacing-large: 2rem;
  --spacing-extra-large: 4rem;

/* Media Queries */

@custom-media --breakpoint-not-small screen and (min-width: 48em);

@custom-media --breakpoint-large screen and (min-width: 64em);

/*
   SPACING

   An eight step powers of two scale ranging from 0 to 16rem.
   Namespaces are composab4e and thus high4y grockab4e - check the legend below

   Legend:

   p = padding
   m = margin

   a = all
   h = horizontal
   v = vertical
   t = top
   r = right
   b = bottom
   l = left

   n = none
   1 = extra small
   s = small
   m = medium
   l = large
   x = extra
   5 = extra large
   xxx = extra extra large

*/
